```javascript
// lib/bidi-detect.js
// Pure ES module — no DOM, no Chrome APIs. Importable by Jest in Node/jsdom.

// U+0590-05FF: Hebrew block (letters, nikkud, cantillation, punctuation)
// U+FB1D-FB4F: Hebrew Presentation Forms (legacy word processor output)
const HEBREW_RE = /[\u0590-\u05FF\uFB1D-\uFB4F]/;

// Subset: actual Hebrew letters and presentation forms (used in 30% threshold)
// U+05D0-05EA: Hebrew letters (alef to tav)
// U+05F0-05F4: Hebrew ligatures/punctuation letters
// U+FB1D-FB4F: Presentation Forms
const HEBREW_LETTER_RE = /[\u05D0-\u05EA\u05F0-\u05F4\uFB1D-\uFB4F]/;

// Characters to skip in first-strong-character pass:
// \d: ASCII digits (0-9)
// \u0021-\u002F: ASCII punctuation (!"#$%&'()*+,-./)
// \u003A-\u0040: ASCII punctuation (:;<=>?@)
// \u005B-\u0060: ASCII punctuation ([\]^_`)
// \u007B-\u00BF: ASCII/Latin-1 non-letter characters ({|}~, DEL, C1 controls, non-breaking chars)
// \uD800-\uDFFF: Surrogate pairs (emoji halves — spread([...text]) handles them but test anyway)
const SKIP_RE = /[\d\u0021-\u002F\u003A-\u0040\u005B-\u0060\u007B-\u00BF]|[\uD800-\uDFFF]/;

export function detectDirection(text) {
  const chars = [...text]; // spread splits at code-point boundaries, not UTF-16 code-unit boundaries

  // Pass 1: first-strong-character algorithm (Unicode BiDi UAX#9 simplified for Hebrew)
  for (const ch of chars) {
    if (SKIP_RE.test(ch)) continue;
    if (HEBREW_RE.test(ch)) return 'rtl';
    return 'ltr'; // first non-neutral, non-Hebrew character → LTR
  }

  // Pass 2: mixed-content 30% Hebrew threshold
  // Only reached when all characters were neutral in Pass 1
  const letterCount = chars.filter(ch => /\p{L}/u.test(ch)).length;
  if (letterCount === 0) return 'ltr'; // no letters at all → LTR default
  const hebrewCount = chars.filter(ch => HEBREW_LETTER_RE.test(ch)).length;
  return hebrewCount / letterCount >= 0.30 ? 'rtl' : 'ltr';
}

export function isExemptElement(el) {
  const tag = el.tagName?.toLowerCase();
  // Code and preformatted text: always LTR
  if (['code', 'pre', 'kbd', 'samp'].includes(tag)) return true;
  // Math rendering classes: direction must be LTR for formula layout
  if (el.matches?.('.katex, .math, [class*="math"], [class*="latex"]')) return true;
  // Ancestor check: element inside math or code context inherits that context
  // This prevents KaTeX inheritance corruption (Pitfall 13 from RESEARCH.md)
  if (el.closest?.('.katex, .math, pre, code')) return true;
  // Content-based exemptions
  const text = el.textContent?.trim() ?? '';
  if (/^(https?|ftp|file):\/\//i.test(text)) return true; // URL
  if (/^\/[a-z0-9._\-/]+$/i.test(text)) return true; // file path (/usr/local/bin)
  return false;
}

export function getFirstSubstantiveText(el) {
  // Walk text nodes in document order using TreeWalker
  // Handles: <li><strong>label</strong> Hebrew text</li>
  // Returns the first non-whitespace text found anywhere in the element tree
  const walker = document.createTreeWalker(el, NodeFilter.SHOW_TEXT);
  let node;
  while ((node = walker.nextNode())) {
    const text = node.textContent.trim();
    if (text.length > 0) return text;
  }
  return el.textContent?.trim() ?? '';
}
```
