```javascript
// content.js
// Phase 1: Per-element RTL style application and detection.
// MutationObserver integration added in Phase 1 Plan 03.
import { detectDirection, isExemptElement, getFirstSubstantiveText } from './lib/bidi-detect.js';

export const MARKER = 'data-hrtl-processed';
export const DEBOUNCE_MS = 100;

/**
 * Apply or clear RTL/LTR inline styles on an element.
 * Always marks element with data-hrtl-processed.
 *
 * RTL: sets direction:rtl, text-align:right. For <li>: also sets list-style-position:inside
 *      (prevents bullet/number from rendering outside the element boundary — Pitfall 9).
 * LTR: clears all three properties (not sets to 'ltr'/'left'/'outside') to preserve page defaults.
 *
 * @param {Element} el
 * @param {'rtl'|'ltr'} dir
 */
export function applyDirection(el, dir) {
  if (dir === 'rtl') {
    el.style.direction = 'rtl';
    el.style.textAlign = 'right';
    // Fix list bullet visibility (ENG-07 / Pitfall 9):
    // When <li> receives direction:rtl, bullets render outside the element boundary
    // unless list-style-position:inside is explicitly set.
    if (el.tagName === 'LI') {
      el.style.listStylePosition = 'inside';
    }
  } else {
    // Clear rather than set 'ltr' — avoids overriding the site's intended defaults
    el.style.direction = '';
    el.style.textAlign = '';
    el.style.listStylePosition = '';
  }
  // Mark as processed. MutationObserver must NOT set attributes:true — this setAttribute
  // would trigger infinite loops if attributes were observed (Pitfall 1 from RESEARCH.md).
  el.setAttribute(MARKER, '1');
}

/**
 * Determine and apply text direction to a single element.
 *
 * Processing order:
 * 1. If forceRTL is true on selectorConfig: apply RTL immediately (bypasses detection AND exemption).
 *    Per ENG-06: "skips detection and always applies RTL". User explicitly requested this selector
 *    to always be RTL — exemption logic must not veto the user's explicit choice.
 * 2. If element is exempt (code, pre, .katex, URL-only content, etc.): return without touching it.
 * 3. Otherwise: detect direction from first substantive text content, then apply.
 *
 * @param {Element} el
 * @param {{ forceRTL?: boolean } | null} selectorConfig
 */
export function processElement(el, selectorConfig) {
  // forceRTL: skip both detection and exemption (ENG-06)
  if (selectorConfig?.forceRTL) {
    applyDirection(el, 'rtl');
    return;
  }

  // LTR preservation: exempt elements must not receive RTL styles (ENG-04)
  if (isExemptElement(el)) return;

  // Detect from first substantive text (handles inline wrappers — ENG-08)
  const text = getFirstSubstantiveText(el);
  const dir = detectDirection(text);
  applyDirection(el, dir);
}

// ---- MutationObserver stubs — implemented in Plan 03 ----

let _observer = null;

/**
 * Start observing document.body for DOM mutations.
 * Stub: Plan 03 implements the full MutationObserver with debounce.
 * @param {string[]} selectors
 * @param {{ forceRTL?: boolean } | null} selectorConfig
 */
export function startObserver(selectors, selectorConfig) {
  // Plan 03 implementation
}

/**
 * Disconnect the MutationObserver.
 * Stub: Plan 03 implements.
 */
export function stopObserver() {
  if (_observer) {
    _observer.disconnect();
    _observer = null;
  }
}
```
