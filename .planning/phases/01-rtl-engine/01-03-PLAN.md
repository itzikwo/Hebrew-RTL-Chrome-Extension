```javascript
// Add these module-level variables below the existing MARKER and DEBOUNCE_MS constants:
let _observer = null;
let _debounceTimer = null;
const _pendingNodes = new Set();

export function startObserver(selectors, selectorConfig) {
  // Disconnect any existing observer before creating a new one
  if (_observer) {
    _observer.disconnect();
    _observer = null;
  }

  _observer = new MutationObserver((mutations) => {
    for (const mutation of mutations) {
      // childList: new elements added to the DOM
      if (mutation.type === 'childList') {
        mutation.addedNodes.forEach(node => {
          if (node.nodeType === Node.ELEMENT_NODE && !node.hasAttribute(MARKER)) {
            _pendingNodes.add(node);
          }
        });
      }

      // characterData: text content of an existing element changed
      // This catches streaming content where characters are added to an existing text node
      // (ENG-05 locked decision: re-evaluate on characterData, bidirectional)
      if (mutation.type === 'characterData' && mutation.target.parentElement) {
        const el = mutation.target.parentElement;
        // Remove marker to allow re-evaluation (bidirectional — LTR→RTL or RTL→LTR)
        el.removeAttribute(MARKER);
        _pendingNodes.add(el);
      }
    }

    // Debounce: batch all mutations within the window into one processing call
    // Re-evaluation runs only within this debounce batch — no separate throttle
    clearTimeout(_debounceTimer);
    _debounceTimer = setTimeout(() => {
      const nodes = [..._pendingNodes];
      _pendingNodes.clear();
      for (const node of nodes) {
        // Skip if already processed (another mutation may have already handled it in this batch)
        if (node.hasAttribute(MARKER)) continue;
        processElement(node, selectorConfig);
        // Also process unprocessed descendants
        node.querySelectorAll?.(`*:not([${MARKER}])`).forEach(child => {
          processElement(child, selectorConfig);
        });
      }
    }, DEBOUNCE_MS);
  });

  // CRITICAL: attributes:true MUST NOT be set.
  // setAttribute(MARKER) would trigger attribute mutations → infinite loop (Pitfall 1).
  _observer.observe(document.body, {
    childList: true,
    subtree: true,
    characterData: true
    // attributes: true  ← intentionally omitted
  });
}

export function stopObserver() {
  clearTimeout(_debounceTimer);
  _pendingNodes.clear();
  if (_observer) {
    _observer.disconnect();
    _observer = null;
  }
}
```
