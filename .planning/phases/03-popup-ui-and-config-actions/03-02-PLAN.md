const actionsBtn = document.getElementById('actions-btn');
const actionsMenu = document.getElementById('actions-menu');

actionsBtn.addEventListener('click', (e) => {
  e.stopPropagation();

document.addEventListener('click', (e) => {

    actionsMenu.hidden = true;

  }

2. **Action handlers** — delegate from actions-menu clicks:
```javascript
actionsMenu.addEventListener('click', async (e) => {
  const action = e.target.closest('[data-action]')?.dataset.action;
  if (!action) return;
  actionsMenu.hidden = true;

  if (action === 'export') {
    const allConfigs = await getAllConfigs();
    const json = JSON.stringify(allConfigs, null, 2);
    const blob = new Blob([json], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    const date = new Date().toISOString().slice(0, 10);
    a.href = url;
    a.download = `hebrew-rtl-config-${date}.json`;
    a.click();
    URL.revokeObjectURL(url);
  }

  if (action === 'delete-all') {
    const confirmDialog = document.getElementById('confirm-dialog');
    const confirmText = document.getElementById('confirm-text');
    confirmText.textContent = `Delete all selectors for ${hostname}? This cannot be undone.`;
    confirmDialog.hidden = false;
  }

  if (action === 'shortcuts') {
    chrome.tabs.create({ url: 'chrome://extensions/shortcuts' });
  }
});
```

3. **Confirm dialog handlers:**
```javascript
document.getElementById('confirm-ok').addEventListener('click', async () => {
  config.selectors = [];
  await setDomainConfig(hostname, config);
  document.getElementById('confirm-dialog').hidden = true;
  renderPopup(hostname, config, tabId);
});

document.getElementById('confirm-cancel').addEventListener('click', () => {
  document.getElementById('confirm-dialog').hidden = true;
});
```

CRITICAL: Do NOT use `window.confirm()` — it is blocked in Chrome extension popups and returns undefined. The custom HTML confirm-dialog is the required approach.
